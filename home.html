function getYoutubeId(value){ const raw = String(value || '').trim(); if(!raw) return ''; if(!raw.includes('youtube') && !raw.includes('youtu.be') && !raw.includes('/')){ return raw; } try{ const url = new URL(raw); if(url.hostname.includes('youtu.be')){ return url.pathname.replace('/','').split('?')[0]; } if(url.hostname.includes('youtube.com')){ if(url.searchParams.get('v')){ return url.searchParams.get('v'); } if(url.pathname.includes('/shorts/')){ return url.pathname.split('/shorts/')[1]?.split('/')[0] || ''; } if(url.pathname.includes('/embed/')){ return url.pathname.split('/embed/')[1]?.split('/')[0] || ''; } } }catch(e){ return raw; } return ''; } function renderCards(data){ const home = data.home || {}; const youtubeId = getYoutubeId(home.sermon_link); const sermonEmbed = youtubeId ? `https://www.youtube.com/embed/${youtubeId}` : ''; const noticeTitle = home.notice_title || '오늘의 안내'; const noticeBody = home.notice_body || '예배와 모임 안내는 교회 공지 채널을 확인해주세요.'; const sermonTitle = home.sermon_title || '이번 주 말씀'; const sermonBody = youtubeId ? '아래에서 이번 주 말씀 영상을 바로 확인할 수 있습니다.' : '설교 영상 또는 말씀 링크가 준비 중입니다.'; const newFamilyText = home.new_family_text || '처음 방문하셨다면 안내위원 또는 담당 리더에게 말씀해주세요.'; const offeringText = home.offering_text || '온라인 헌금 안내는 교회 공식 안내를 확인해주세요.'; const bulletinLink = home.bulletin_link || ''; const cards = []; cards.push(`

${escapeHtml(noticeTitle)}

${escapeHtml(noticeBody)}

`); cards.push(`

${escapeHtml(sermonTitle)}

${escapeHtml(sermonBody)}

${ sermonEmbed ? `
` : '' }
`); if(bulletinLink){ cards.push(`

주보 보기

이번 주 주보를 확인할 수 있습니다.
